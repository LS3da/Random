乱数生成プログラム (Batch)

@echo off
cls
echo Random Creater v1.21
title Random Create
echo To stop this program, press the close button in the upper-right corner.
pause > NUL
echo.
set /p "Ranif=Please select a range (number only): "
set /p yn_check="Create Log Files? (y/n)"
IF %yn_check%==y (
  set /a Count = 0
  goto StartLog
) ELSE (
  goto StartNLog
)

echo ERROR DEVELOPMENT BUGS!(NO END SUPPORT)
pause
exit


:StartLog
cls
set /a Count = %Count%+1
set /a RandomCounter = %RANDOM% * %Ranif% / 32767
echo %Count%. %RandomCounter% >> RandomLog.txt
echo %RandomCounter%
echo.
echo Press any key to generate again...
echo.
pause > NUL
goto StartLog


:StartNLog
cls
set /a RandomCounter = %RANDOM% * %Ranif% / 32767
echo %RandomCounter%
echo.
echo Press any key to generate again...
echo.
pause > NUL
goto StartNLog
