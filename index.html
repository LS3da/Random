乱数生成プログラム (Batch)

@echo off
cls
echo Random Creater v1.21
title Random Create
echo To stop this program, press the close button in the upper-right corner.
pause > NUL
echo.
set /p "Ranif=Please select a range (number only): "
set /p yn_check="Create Log Files? (y/n)"
IF %yn_check%==y (
  set /a Count = 0
  goto StartLog
) ELSE (
  goto StartNLog
)

echo ERROR DEVELOPMENT BUGS!(NO END SUPPORT)
pause
exit


:StartLog
cls
set /a Count = %Count%+1
set /a RandomCounter = %RANDOM% * %Ranif% / 32767
echo %Count%. %RandomCounter% >> RandomLog.txt
echo %RandomCounter%
echo.
echo Press any key to generate again...
echo.
pause > NUL
goto StartLog


:StartNLog
cls
set /a RandomCounter = %RANDOM% * %Ranif% / 32767
echo %RandomCounter%
echo.
echo Press any key to generate again...
echo.
pause > NUL
goto StartNLog

ルーレット (Batch)

@echo off
chcp 65001
cls
goto setup1

:setup1
echo 参加者数を教えてください。
echo.
set /p "Sankasha=Count: "

for /l %%i in (1, 1, %Sankasha%) do (
    set /p "SankaName%%i=参加者%%i人目: "
)

echo.
echo 確認が取れました。ルーレットを開始するには何かキーを押してください...
pause > NUL
goto ruretto

:ruretto
cls
setlocal EnableDelayedExpansion
set /a rurettokekka = %RANDOM% %% %Sankasha%+1
echo 抽選結果発表！
echo.
timeout /t 1 >nul
echo 当選者は...
echo.
timeout /t 3 >nul
echo !SankaName%rurettokekka%!だ！おめでとう！
echo.
echo 何かキーを押すと閉じれます。
pause > NUL
exit
